select
	Products.id,
	Category.name,
	Sub_Category.name,
	Rarity.name,
	Products.name,
	Products.quantity,
	Products.price
from products
left OUTER JOIN Rarity
ON products.rarityid = Rarity.id
left OUTER JOIN Category
ON products.sub_categoryid = Category.id
left OUTER JOIN Sub_Category
ON products.sub_categoryid = Sub_Category.id;
